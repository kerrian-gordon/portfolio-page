// ── Dashboard Tab ─────────────────────────────────────────────────────────────
function DashboardTab({schools,vacancies}) {
  const total    = schools.reduce((a,s)=>a+s.capacity,0);
  const enrolled = schools.reduce((a,s)=>a+s.enrolled,0);
  const waitlist = schools.reduce((a,s)=>a+s.waitlist,0);
  const programs = schools.filter(s=>s.hasProgram).length;
  const openVac  = vacancies.filter(v=>v.status==="open").reduce((a,v)=>a+v.seats,0);
  const byBorough = BOROUGHS.map(b=>{
    const bs=schools.filter(s=>s.borough===b);
    return {borough:b,capacity:bs.reduce((a,s)=>a+s.capacity,0),enrolled:bs.reduce((a,s)=>a+s.enrolled,0),waitlist:bs.reduce((a,s)=>a+s.waitlist,0),programs:bs.filter(s=>s.hasProgram).length};
  });
  return (
    <div style={{display:"flex",flexDirection:"column",gap:24}}>
      <div style={{display:"flex",gap:16,flexWrap:"wrap"}}>
        <StatCard label="Total 3K Seats"      value={total.toLocaleString()}    sub="Across all programs"          accent="#38bdf8"/>
        <StatCard label="Currently Enrolled"  value={enrolled.toLocaleString()} sub={`${pct(enrolled,total)}% utilization`} accent="#a78bfa"/>
        <StatCard label="Waitlisted Families" value={waitlist.toLocaleString()} sub="Seeking placement"             accent="#f97316"/>
        <StatCard label="Active Programs"     value={programs}                  sub={`of ${schools.length} schools`} accent="#22c55e"/>

        <StatCard label="Via Live Feed"    value={vacancies.filter(v=>v.source==="auto").length}   sub="Auto-detected"   accent="#a78bfa"/>
        <StatCard label="Via Manual Entry" value={vacancies.filter(v=>v.source==="manual").length} sub="Staff reported"   accent="#f97316"/>
      </div>
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"center"}}>
        <div style={{fontSize:15,fontWeight:700,color:"#f1f5f9"}}>Live Vacancy Feed</div>
        <div style={{display:"flex",alignItems:"center",gap:10}}>
          <span style={{fontSize:12,color:"#64748b"}}>Simulated feed:</span>
          <button onClick={()=>setLiveEnabled(p=>!p)}
            style={{background:liveEnabled?"rgba(34,197,94,0.15)":"rgba(255,255,255,0.06)",border:`1px solid ${liveEnabled?"rgba(34,197,94,0.4)":"rgba(255,255,255,0.1)"}`,borderRadius:99,padding:"5px 16px",color:liveEnabled?"#22c55e":"#64748b",fontSize:11,fontWeight:700,cursor:"pointer",display:"flex",alignItems:"center",gap:6}}>

            {liveEnabled?"ON":"OFF"}
          </button>
        </div>
      </div>
      {open.length===0 && (
        <div style={{padding:32,textAlign:"center",color:"#475569",fontSize:13,background:"rgba(255,255,255,0.02)",borderRadius:12,border:"1px solid rgba(255,255,255,0.05)"}}>
          No open vacancies right now. Enable the live feed or log one manually.
        </div>
      )}
      {open.map(v=>(

// ── Seat Reassignment Tab ─────────────────────────────────────────────────────
function ReassignmentTab({schools,vacancies,onFill}) {
  const [families,setFamilies] = useState(INITIAL_FAMILIES);
  const [placements,setPlacements] = useState({});
  const [selected,setSelected] = useState(null);

  const effectiveAvail = s => {
    const vacSeats = vacancies.filter(v=>v.schoolId===s.id&&v.status==="open").reduce((a,v)=>a+v.seats,0);
    const placed   = Object.values(placements).filter(p=>p.schoolId===s.id).length;
    return Math.max(0, s.capacity - s.enrolled + vacSeats - placed);
  };

  const getSchool = id => schools.find(s=>s.id===id);

  const suggest = fam => {
    for(const id of fam.preferredSchools){ const s=getSchool(id); if(s&&effectiveAvail(s)>0) return s; }
    return schools.filter(s=>s.hasProgram&&effectiveAvail(s)>0).sort((a,b)=>b.score-a.score)[0]||null;
  };

  const place = (famId,school) => {
    setPlacements(p=>({...p,[famId]:{schoolId:school.id,schoolName:school.name}}));
    setFamilies(f=>f.map(fam=>fam.id===famId?{...fam,status:"placed"}:fam));
  };

  const autoPlace = () => {
    const np={...placements};
    const nf=families.map(fam=>{
      if(fam.status==="placed") return fam;
      const s=suggest(fam);
      if(s){np[fam.id]={schoolId:s.id,schoolName:s.name};return{...fam,status:"placed"};}
      return fam;
    });
    setPlacements(np); setFamilies(nf);
  };

  const placed = families.filter(f=>f.status==="placed").length;
  const openVac = vacancies.filter(v=>v.status==="open");

  return (
    <div style={{display:"flex",gap:20,flexWrap:"wrap"}}>
      <div style={{flex:1,minWidth:300,display:"flex",flexDirection:"column",gap:16}}>
        {openVac.length>0 && (
          <div style={{background:"rgba(56,189,248,0.06)",border:"1px solid rgba(56,189,248,0.25)",borderRadius:12,padding:"14px 16px"}}>
            <div style={{fontSize:13,fontWeight:700,color:"#38bdf8",marginBottom:8}}>⚡ {openVac.length} Open Vacanc{openVac.length>1?"ies":"y"} — Fill Now</div>
            {openVac.map(v=>{ const s=getSchool(v.schoolId); if(!s) return null;
              const nextFam=families.find(f=>f.status==="unplaced");
              return (
                <div key={v.id} style={{display:"flex",justifyContent:"space-between",alignItems:"center",padding:"8px 0",borderBottom:"1px solid rgba(56,189,248,0.1)"}}>
                  <div>
                    <span style={{fontSize:12,fontWeight:600,color:"#f1f5f9"}}>{s.name.split(" ").slice(0,3).join(" ")}</span>
                    <span style={{fontSize:11,color:"#64748b",marginLeft:8}}>{v.seats} seat{v.seats>1?"s":""} · {v.reason}</span>
                  </div>
                  <button onClick={()=>{onFill(v.id);if(nextFam)place(nextFam.id,s);}}
                    style={{background:"rgba(34,197,94,0.15)",border:"1px solid rgba(34,197,94,0.3)",color:"#22c55e",padding:"5px 12px",borderRadius:6,fontSize:11,fontWeight:700,cursor:"pointer"}}>
                    Fill →
                  </button>
                </div>
              );
            })}
          </div>
        )}
        <div style={{display:"flex",justifyContent:"space-between",alignItems:"center"}}>
          <div>
            <div style={{fontSize:16,fontWeight:700,color:"#f1f5f9"}}>Waitlist Families</div>
            <div style={{fontSize:12,color:"#64748b",marginTop:2}}>{placed} of {families.length} placed this cycle</div>
          </div>
          <button onClick={autoPlace} style={{background:"linear-gradient(135deg,#38bdf8,#818cf8)",border:"none",color:"#fff",padding:"10px 20px",borderRadius:8,fontSize:12,fontWeight:700,cursor:"pointer"}}>⚡ Auto-Optimize All</button>
        </div>
        <div style={{height:6,background:"rgba(255,255,255,0.06)",borderRadius:99}}>
          <div style={{width:`${pct(placed,families.length)}%`,height:"100%",background:"linear-gradient(90deg,#38bdf8,#818cf8)",borderRadius:99,transition:"width 0.6s ease"}}/>
        </div>
        {families.map(fam=>{
          const sugg=suggest(fam), isPlaced=fam.status==="placed", placement=placements[fam.id];
          return (
            <div key={fam.id} onClick={()=>setSelected(selected===fam.id?null:fam.id)}
              style={{background:selected===fam.id?"rgba(56,189,248,0.06)":"rgba(255,255,255,0.03)",border:`1px solid ${selected===fam.id?"rgba(56,189,248,0.3)":"rgba(255,255,255,0.07)"}`,borderRadius:12,padding:"14px 16px",cursor:"pointer",transition:"all 0.2s"}}>
              <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start"}}>

                    {isPlaced && <Badge label="✓ Placed" color="#22c55e" bg="rgba(34,197,94,0.1)"/>}
                  </div>
                  <div style={{fontSize:11,color:"#64748b",marginTop:4}}>{fam.name} · {fam.distance}mi · Income: {fam.income}</div>
                </div>
                {!isPlaced&&sugg&&(
                  <button onClick={e=>{e.stopPropagation();place(fam.id,sugg);}}

                  {isPlaced ? (

                  ) : (
                    <div>
                      <div style={{fontSize:11,color:"#64748b",marginBottom:8}}>Preferred schools &amp; availability:</div>
                      {fam.preferredSchools.map(sid=>{
                        const s=getSchool(sid); if(!s) return null;
                        const avail=effectiveAvail(s);
                        return (
                          <div key={sid} style={{display:"flex",justifyContent:"space-between",alignItems:"center",padding:"6px 0"}}>
                            <span style={{fontSize:12,color:"#cbd5e1"}}>{s.name.split(" ").slice(0,4).join(" ")}</span>
                            <div style={{display:"flex",gap:8,alignItems:"center"}}>
                              <Badge label={avail>0?`${avail} open`:"Full"} color={avail>0?"#22c55e":"#ef4444"} bg={avail>0?"rgba(34,197,94,0.1)":"rgba(239,68,68,0.1)"}/>
                              {avail>0&&<button onClick={e=>{e.stopPropagation();place(fam.id,s);}} style={{background:"rgba(56,189,248,0.15)",border:"none",color:"#38bdf8",padding:"4px 10px",borderRadius:5,fontSize:10,cursor:"pointer",fontWeight:700}}>Select</button>}
                            </div>
                          </div>
                        );
                      })}
                    </div>
                  )}
                </div>
              )}
            </div>
          );
        })}
      </div>
      <div style={{width:240,flexShrink:0}}>
        <div style={{background:"rgba(255,255,255,0.03)",border:"1px solid rgba(255,255,255,0.07)",borderRadius:14,padding:20,position:"sticky",top:80}}>
          <div style={{fontSize:13,fontWeight:600,color:"#cbd5e1",marginBottom:16}}>Cycle Summary</div>
          {[["Families in Queue",families.length,"#f1f5f9"],["High Priority",families.filter(f=>f.priority==="high").length,"#f97316"],["Placed",placed,"#22c55e"],["Pending",families.length-placed,"#94a3b8"],["Vacancy Seats Open",openVac.reduce((a,v)=>a+v.seats,0),"#38bdf8"]].map(([l,v,c])=>(

  );
}

// ── Capacity Tab ──────────────────────────────────────────────────────────────
function CapacityTab({schools,vacancies}) {
  const [bFilter,setBFilter] = useState("All");
  const [sort,setSort]       = useState("utilization");
  const [search,setSearch]   = useState("");
  const vacFor = id => vacancies.filter(v=>v.schoolId===id&&v.status==="open").reduce((a,v)=>a+v.seats,0);
  const rows = schools
    .filter(s=>bFilter==="All"||s.borough===bFilter)
    .filter(s=>s.name.toLowerCase().includes(search.toLowerCase()))
    .sort((a,b)=>{
      if(sort==="utilization") return pct(b.enrolled,b.capacity)-pct(a.enrolled,a.capacity);
      if(sort==="waitlist")    return b.waitlist-a.waitlist;
      if(sort==="available")   return (b.capacity-b.enrolled)-(a.capacity-a.enrolled);
      return a.name.localeCompare(b.name);
    });
  return (
    <div style={{display:"flex",flexDirection:"column",gap:16}}>
      <div style={{display:"flex",gap:10,flexWrap:"wrap",alignItems:"center"}}>
        <input value={search} onChange={e=>setSearch(e.target.value)} placeholder="Search schools…"

// ── Demand Analytics Tab ──────────────────────────────────────────────────────
function DemandTab({schools}) {
  const totalWl = schools.reduce((a,s)=>a+s.waitlist,0);
  const highD   = schools.filter(s=>s.waitlist>15);
  const byB = BOROUGHS.map(b=>{
    const bs=schools.filter(s=>s.borough===b);
    const wl=bs.reduce((a,s)=>a+s.waitlist,0);
    const gap=bs.reduce((a,s)=>a+Math.max(0,s.capacity-s.enrolled),0);
    return{borough:b,waitlist:wl,gap,net:wl-gap};
  });
  const maxWl = Math.max(...byB.map(b=>b.waitlist));
  return (
    <div style={{display:"flex",flexDirection:"column",gap:20}}>
      <div style={{display:"flex",gap:16,flexWrap:"wrap"}}>
        <StatCard label="Total Waitlisted"    value={totalWl}        sub="Seeking 3K placement"  accent="#f97316"/>
        <StatCard label="High-Demand Schools" value={highD.length}   sub="15+ on waitlist"        accent="#ef4444"/>
        <StatCard label="Unmet Demand Gap"    value={`${Math.max(0,totalWl-schools.reduce((a,s)=>a+Math.max(0,s.capacity-s.enrolled),0))} seats`} sub="Estimated shortfall" accent="#a78bfa"/>
        <StatCard label="Demand Coverage"     value={`${pct(schools.filter(s=>s.hasProgram).length,schools.length)}%`} sub="Schools with programs" accent="#22c55e"/>
      </div>
      <div style={{background:"rgba(255,255,255,0.03)",border:"1px solid rgba(255,255,255,0.07)",borderRadius:14,padding:20}}>
        <div style={{fontSize:13,fontWeight:600,color:"#cbd5e1",marginBottom:20}}>Waitlist Demand by Borough</div>
        {byB.sort((a,b)=>b.waitlist-a.waitlist).map(b=>(
          <div key={b.borough} style={{marginBottom:14}}>
            <div style={{display:"flex",justifyContent:"space-between",marginBottom:6}}>
              <span style={{fontSize:13,fontWeight:600,color:"#e2e8f0"}}>{b.borough}</span>
              <div style={{display:"flex",gap:12}}>
                <span style={{fontSize:11,color:"#64748b"}}>{b.gap} available</span>
                <span style={{fontSize:13,color:b.waitlist>0?"#f97316":"#64748b",fontWeight:700}}>{b.waitlist} waiting</span>
              </div>
            </div>
            <div style={{height:10,background:"rgba(255,255,255,0.05)",borderRadius:99,overflow:"hidden"}}>
              <div style={{height:"100%",width:`${(b.waitlist/maxWl)*100}%`,background:b.net>10?"linear-gradient(90deg,#ef4444,#f97316)":b.net>0?"linear-gradient(90deg,#f97316,#fbbf24)":"linear-gradient(90deg,#22c55e,#38bdf8)",borderRadius:99,transition:"width 0.8s ease"}}/>
            </div>
            {b.net>0&&<div style={{fontSize:11,color:"#f97316",marginTop:4}}>⚠ {b.net} families exceed available seats — expansion needed</div>}
          </div>
        ))}
      </div>
      <div style={{display:"grid",gridTemplateColumns:"repeat(auto-fill,minmax(280px,1fr))",gap:16}}>
        <div style={{background:"rgba(255,255,255,0.03)",border:"1px solid rgba(255,255,255,0.07)",borderRadius:14,padding:20}}>
          <div style={{fontSize:13,fontWeight:600,color:"#cbd5e1",marginBottom:16}}>Top Demand Schools</div>
          {schools.filter(s=>s.waitlist>0).sort((a,b)=>b.waitlist-a.waitlist).slice(0,8).map((s,i)=>(
            <div key={s.id} style={{display:"flex",alignItems:"center",gap:12,padding:"7px 0",borderBottom:"1px solid rgba(255,255,255,0.04)"}}>
              <span style={{fontSize:11,color:"#475569",minWidth:16}}>{i+1}</span>
              <div style={{flex:1}}>
                <div style={{fontSize:12,color:"#e2e8f0",fontWeight:500}}>{s.name.split(" ").slice(0,3).join(" ")}</div>
                <div style={{fontSize:10,color:"#475569"}}>{s.borough}</div>
              </div>
              <div style={{textAlign:"right"}}>
                <div style={{fontSize:14,fontWeight:700,color:"#f97316"}}>{s.waitlist}</div>
                <div style={{fontSize:9,color:"#475569",letterSpacing:"0.05em"}}>WAITING</div>
              </div>
            </div>
          ))}
        </div>
        <div style={{background:"rgba(255,255,255,0.03)",border:"1px solid rgba(255,255,255,0.07)",borderRadius:14,padding:20}}>
          <div style={{fontSize:13,fontWeight:600,color:"#cbd5e1",marginBottom:16}}>Income Equity Analysis</div>
          {["low","mixed"].map(inc=>{
            const is=schools.filter(s=>s.income===inc);
            const iw=is.reduce((a,s)=>a+s.waitlist,0);
            const ip=is.filter(s=>s.hasProgram).length;
            const iu=Math.round(is.reduce((a,s)=>a+pct(s.enrolled,s.capacity),0)/is.length);
            return (
              <div key={inc} style={{marginBottom:20}}>
                <div style={{display:"flex",justifyContent:"space-between",marginBottom:8}}>
                  <Badge label={inc==="low"?"Low-Income Areas":"Mixed-Income Areas"} color={inc==="low"?"#a78bfa":"#38bdf8"} bg={inc==="low"?"rgba(167,139,250,0.1)":"rgba(56,189,248,0.1)"}/>
                  <span style={{fontSize:11,color:"#64748b"}}>{is.length} schools</span>
                </div>
                {[["Programs active",ip,is.length],["Avg. utilization",iu,100,"%"],["Total waitlist",iw,null]].map(([l,v,t,u])=>(
                  <div key={l} style={{display:"flex",justifyContent:"space-between",padding:"5px 0",borderBottom:"1px solid rgba(255,255,255,0.04)"}}>
                    <span style={{fontSize:11,color:"#64748b"}}>{l}</span>
                    <span style={{fontSize:12,fontWeight:600,color:"#f1f5f9"}}>{v}{u||""}{t?` / ${t}`:""}</span>
                  </div>
                ))}
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}

const TABS = [
  {id:"dashboard",   label:"Overview",         icon:"◈"},
  {id:"vacancies",   label:"Vacancy Feed",      icon:"⚡"},
  {id:"reassign",    label:"Seat Reassignment", icon:"⟳"},
  {id:"expansion",   label:"Expansion Planner", icon:"⊕"},
  {id:"capacity",    label:"School Capacity",   icon:"▦"},
  {id:"demand",      label:"Demand Analytics",  icon:"◎"},
];

function App() {
  const [tab,          setTab]         = useState("dashboard");
  const [vacancies,    setVacancies]   = useState([]);
  const [liveEnabled,  setLiveEnabled] = useState(true);
  const [showModal,    setShowModal]   = useState(false);
  const [bellPulse,    setBellPulse]   = useState(false);
  const timerRef = useRef(null);
  const schools  = INITIAL_SCHOOLS;

  const openCount = vacancies.filter(v=>v.status==="open").length;

  useEffect(()=>{
    if(!liveEnabled){clearInterval(timerRef.current);return;}
    timerRef.current = setInterval(()=>{
      const active = INITIAL_SCHOOLS.filter(s=>s.hasProgram&&s.enrolled>0);
      const s      = active[Math.floor(Math.random()*active.length)];
      const v = {id:++vacId,schoolId:s.id,seats:Math.random()<0.7?1:2,reason:VACANCY_REASONS[Math.floor(Math.random()*VACANCY_REASONS.length)],source:"auto",status:"open",time:nowTs()};
      setVacancies(p=>[v,...p.slice(0,29)]);
      setBellPulse(true); setTimeout(()=>setBellPulse(false),900);
    },12000);
    return ()=>clearInterval(timerRef.current);
  },[liveEnabled]);

  const addManual = ({schoolId,seats,reason}) => {
    const v={id:++vacId,schoolId,seats,reason,source:"manual",status:"open",time:nowTs()};
    setVacancies(p=>[v,...p]);
    setBellPulse(true); setTimeout(()=>setBellPulse(false),900);
  };
  const dismiss = id => setVacancies(p=>p.filter(v=>v.id!==id));
  const fill    = id => setVacancies(p=>p.map(v=>v.id===id?{...v,status:"filled",time:nowTs()}:v));

  return (
    <div style={{minHeight:"100vh",background:"#080c14",color:"#f1f5f9",fontFamily:"'DM Sans',sans-serif"}}>

      {/* Header */}
      <div style={{borderBottom:"1px solid rgba(255,255,255,0.06)",background:"rgba(8,12,20,0.97)",backdropFilter:"blur(20px)",position:"sticky",top:0,zIndex:100}}>
        <div style={{maxWidth:1280,margin:"0 auto",padding:"0 20px",display:"flex",alignItems:"center",justifyContent:"space-between",height:62}}>
          <div style={{display:"flex",alignItems:"center",gap:12}}>
            <div style={{width:32,height:32,borderRadius:8,background:"linear-gradient(135deg,#38bdf8,#818cf8)",display:"flex",alignItems:"center",justifyContent:"center",fontSize:16,fontWeight:900,color:"#fff",flexShrink:0}}>3</div>
            <div>
              <div style={{fontSize:15,fontWeight:700,color:"#f1f5f9",letterSpacing:"-0.02em"}}>NYC 3K Optimizer</div>
              <div style={{fontSize:10,color:"#475569",letterSpacing:"0.08em",textTransform:"uppercase"}}>Seat Management &amp; Expansion Platform</div>
            </div>
          </div>
          <div style={{display:"flex",alignItems:"center",gap:8,flexWrap:"wrap",justifyContent:"flex-end"}}>
            {/* Bell */}
            <button onClick={()=>setTab("vacancies")} className={bellPulse?"bell-ring":""} style={{position:"relative",background:"rgba(255,255,255,0.05)",border:"1px solid rgba(255,255,255,0.1)",borderRadius:8,padding:"6px 12px",cursor:"pointer",display:"flex",alignItems:"center",gap:6,color:"#94a3b8",fontSize:13}}>
              🔔 <span style={{fontSize:11}}>Vacancies</span>
              {openCount>0&&<span style={{position:"absolute",top:-5,right:-5,background:"#ef4444",color:"#fff",borderRadius:99,fontSize:10,fontWeight:800,minWidth:18,height:18,display:"flex",alignItems:"center",justifyContent:"center",padding:"0 4px",boxShadow:"0 0 8px rgba(239,68,68,0.6)"}}>{openCount}</span>}
            </button>
            <button onClick={()=>setShowModal(true)} style={{background:"rgba(167,139,250,0.12)",border:"1px solid rgba(167,139,250,0.3)",borderRadius:8,padding:"6px 14px",cursor:"pointer",color:"#a78bfa",fontSize:11,fontWeight:700}}>+ Log Vacancy</button>
            <div style={{display:"flex",alignItems:"center",gap:6}}>
              <div className={liveEnabled?"pulse-dot":""} style={{width:8,height:8,borderRadius:"50%",background:liveEnabled?"#22c55e":"#475569",boxShadow:liveEnabled?"0 0 8px #22c55e":"none"}}/>
              <span style={{fontSize:11,color:"#64748b"}}>Live · SY 2025–26</span>
            </div>
          </div>
        </div>
      </div>

      {/* Nav */}
      <div style={{borderBottom:"1px solid rgba(255,255,255,0.05)",background:"rgba(8,12,20,0.7)",overflowX:"auto"}}>
        <div style={{maxWidth:1280,margin:"0 auto",padding:"0 20px",display:"flex",gap:2,minWidth:"max-content"}}>
          {TABS.map(t=>(
            <button key={t.id} onClick={()=>setTab(t.id)}
              style={{padding:"14px 16px",background:"transparent",border:"none",borderBottom:`2px solid ${tab===t.id?"#38bdf8":"transparent"}`,color:tab===t.id?"#38bdf8":"#475569",fontSize:12,fontWeight:tab===t.id?700:500,cursor:"pointer",display:"flex",alignItems:"center",gap:6,transition:"all 0.2s",letterSpacing:"0.02em",whiteSpace:"nowrap",position:"relative"}}>
              <span style={{fontSize:14}}>{t.icon}</span>{t.label}
              {t.id==="vacancies"&&openCount>0&&<span style={{background:"#ef4444",color:"#fff",borderRadius:99,fontSize:9,fontWeight:800,padding:"1px 5px",marginLeft:2}}>{openCount}</span>}
            </button>
          ))}
        </div>
      </div>

      {/* Content */}
      <div style={{maxWidth:1280,margin:"0 auto",padding:"28px 20px"}} className="tab-content">
        {tab==="dashboard" && <DashboardTab    schools={schools} vacancies={vacancies}/>}
        {tab==="vacancies" && <VacancyTab      schools={schools} vacancies={vacancies} onDismiss={dismiss} onFill={fill} liveEnabled={liveEnabled} setLiveEnabled={setLiveEnabled}/>}
        {tab==="reassign"  && <ReassignmentTab schools={schools} vacancies={vacancies} onFill={fill}/>}
        {tab==="expansion" && <ExpansionTab    schools={schools}/>}
        {tab==="capacity"  && <CapacityTab     schools={schools} vacancies={vacancies}/>}
        {tab==="demand"    && <DemandTab       schools={schools}/>}
      </div>

      {showModal && <ManualVacancyModal schools={schools} onSubmit={addManual} onClose={()=>setShowModal(false)}/>}
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App/>);
</script>
